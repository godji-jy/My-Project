// ── Firebase: ส่งครั้งเดียวเป็น JSON ──────────
if (Firebase.ready() && signupOK) {
  FirebaseJson json;
  json.set("rpm",         rpm);
  json.set("current",     current);
  json.set("vibration",   vibOK ? vibration : -1);
  json.set("temperature", tempOK ? tempC : -999);
  json.set("total",       (int)total);

  if (Firebase.RTDB.setJSON(&fbdo, "/motor", &json)) {
    Serial.println("Uploaded OK ✓");
  } else {
    Serial.println(fbdo.errorReason());
  }
}

volatile unsigned long pulseCount = 0;
volatile unsigned long totalCount = 0;

void IRAM_ATTR hallISR() {
  pulseCount++;
  totalCount++;
}

// อ่านค่าแบบ atomic (กัน race condition)
noInterrupts();
unsigned long count = pulseCount;
pulseCount = 0;
interrupts();

float rpm = count * 60.0;
